194uint32_t getpH(uint32_t volt)
195{
196 uint32_t aux = 14 / 3000; // Factor de conversion, segun la relacion 0V-3V con pH de 0 a 14
197 // se utiliza el valor de voltaje en milivotls (mV)
198 uint32_t pH_aux = volt * aux;

259
